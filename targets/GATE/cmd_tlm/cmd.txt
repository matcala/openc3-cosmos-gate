# Sample definition of CCSDS Space Packet Standard telecommands.
#
# Fields like LENGTH and CHECKSUM are included as an example but not updated.
# The SER_CMD field, on the other hand, gets updated with the bytes returned by 
# the ground-side Aranya Gate, which the flight-side Aranya endpoint can ingest 
# to check the command against the Aranya Policy and execute it locally.
#
# The target device for this example simply receives the telecommands and logs them.


COMMAND UDPTGT ARANYA_EP_NOOP BIG_ENDIAN    "Aranya Endpoint NOOP Command"
  APPEND_PARAMETER CCSDS_STREAMID      16 UINT MIN_UINT16 MAX_UINT16 0x18A0 "CCSDS Packet Identification" BIG_ENDIAN
  APPEND_PARAMETER CCSDS_SEQUENCE      16 UINT MIN_UINT16 MAX_UINT16 0xC000 "CCSDS Packet Sequence Control" BIG_ENDIAN
  APPEND_PARAMETER CCSDS_LENGTH        16 UINT MIN_UINT16 MAX_UINT16 1      "CCSDS Packet Data Length" BIG_ENDIAN
  APPEND_PARAMETER CCSDS_FC            8  UINT MIN_UINT8 MAX_UINT8 1        "CCSDS Command Function Code" 
  APPEND_PARAMETER CCSDS_CHECKSUM      8  UINT MIN_UINT8 MAX_UINT8 0        "CCSDS Command Checksum"

COMMAND UDPTGT ARANYA_EP_EXP1 BIG_ENDIAN    "Capture Small Image with ARDUCAM"
  APPEND_PARAMETER CCSDS_STREAMID      16 UINT MIN_UINT16 MAX_UINT16 0x18A0 "CCSDS Packet Identification" BIG_ENDIAN
  APPEND_PARAMETER CCSDS_SEQUENCE      16 UINT MIN_UINT16 MAX_UINT16 0xC000 "CCSDS Packet Sequence Control" BIG_ENDIAN
  APPEND_PARAMETER CCSDS_LENGTH        16 UINT MIN_UINT16 MAX_UINT16 1      "CCSDS Packet Data Length" BIG_ENDIAN
  APPEND_PARAMETER CCSDS_FC            8  UINT MIN_UINT8 MAX_UINT8 2        "CCSDS Command Function Code"
  APPEND_PARAMETER CCSDS_CHECKSUM      8  UINT MIN_UINT8 MAX_UINT8 0        "CCSDS Command Checksum"
  # Length field 0 allows for a variable-length data block, initialized with zeroes
  APPEND_PARAMETER SER_CMD             0  BLOCK 0x0                         "Serialized Aranya Command"
